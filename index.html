async function cargarEquiposDesdeCSV() { try { setBadge(' CARGANDO'); const url = GITHUB_CSV_URL() + '?t=' + Date.now(); console.log("Cargando CSV desde:", url); const respuesta = await fetch(url); if (!respuesta.ok) { throw new Error(`HTTP ${respuesta.status} - ${respuesta.statusText}`); } const texto = await respuesta.text(); if (!texto || texto.length < 10) { throw new Error('CSV vacío o inválido'); } const lineas = texto.replace(/^\uFEFF/, '').trim().split('\n'); if (lineas.length < 2) { throw new Error('CSV sin datos'); } const encabezados = lineas[0] .split(',') .map(h => h.trim().toLowerCase().replace(/\r/g, '')); const equiposRaw = []; for (let i = 1; i < lineas.length; i++) { const linea = lineas[i].replace(/\r/g,''); if (!linea.trim()) continue; const valores = parseCsvLine(linea); const obj = {}; encabezados.forEach((h, idx) => { obj[h] = (valores[idx] || '').trim(); }); if (obj.id && obj.nombre) equiposRaw.push(obj); } db.equipos = equiposRaw.map(e => ({ id: parseInt(e.id), nombre: e.nombre, plantel: e.plantel || 'Sin plantel', turno: (e.turno || '').toLowerCase().includes('vespertino') ? 'vespertino' : 'matutino', activo: (e.estatus || 'activo').toLowerCase() !== 'inactivo' })); setBadge('● ONLINE', '#064e3b'); document.getElementById('gh-status').innerHTML = `✅ ${db.equipos.length} equipos`; return db.equipos; } catch(err) { console.error('Error CSV DETALLADO:', err); setBadge('● ERROR CSV', '#7f1d1d'); document.getElementById('gh-status').innerHTML = `❌ CSV: ${err.message}`; return []; } }
